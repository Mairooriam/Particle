cmake_minimum_required(VERSION 3.10)
project(playground C)

set(CMAKE_WINDOWS_EXPORT_ALL_SYMBOLS ON)
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -Wall -Wextra -Wpedantic")
set(CMAKE_COLOR_DIAGNOSTICS ON)

if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    add_compile_definitions(SLOW_CODE_ALLOWED)
    add_compile_definitions(ENTITY_POOL_DEVELOPMENT)
endif()

add_library(entity_lib STATIC
    entity/entityPool.c
    entity/entity_types.h
    entity/entityPool_types.h
    entity/entityPool.h
)

target_include_directories(entity_lib PUBLIC entity)

add_executable(alloc alloc.c)

add_executable(main_entitypool main_entitypool.c)
target_link_libraries(main_entitypool entity_lib)

add_executable(bitwise bitstuff.c)
